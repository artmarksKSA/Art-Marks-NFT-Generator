# ============================================ # ARTWORK TO NFT ANIMATOR # Professional NFT Generator # ============================================ # تثبيت: # pip install pillow moviepy numpy opencv-python imageio imageio-ffmpeg from PIL import Image, ImageEnhance, ImageFilter from moviepy.editor import * import numpy as np import cv2 import random # -------------------------------- # تحميل اللوحة # -------------------------------- IMAGE_PATH = "input.jpg" img = Image.open(IMAGE_PATH).convert("RGB") # -------------------------------- # Resize احترافي # -------------------------------- SIZE = 1080 img = img.resize((SIZE, SIZE), Image.LANCZOS) # -------------------------------- # تحسين الجودة بدون تخريب اللوحة # -------------------------------- img = ImageEnhance.Sharpness(img).enhance(1.2) img = ImageEnhance.Contrast(img).enhance(1.08) img = ImageEnhance.Color(img).enhance(1.05) # Glow خفيف glow = img.filter(ImageFilter.GaussianBlur(radius=8)) img = Image.blend(img, glow, 0.08) img.save("art_temp.png") # -------------------------------- # حركة سينمائية ناعمة # -------------------------------- clip = ImageClip("art_temp.png").set_duration(8) def motion(get_frame, t): frame = Image.fromarray(get_frame(t)) zoom = 1 + (t * 0.01) w, h = frame.size nw = int(w * zoom) nh = int(h * zoom) frame = frame.resize((nw, nh), Image.LANCZOS) left = (nw - w) // 2 top = (nh - h) // 2 frame = frame.crop((left, top, left + w, top + h)) return np.array(frame) animated = clip.fl(motion) # -------------------------------- # particles فنية خفيفة # -------------------------------- particles = [] for i in range(40): particles.append({ "x": random.randint(0, SIZE), "y": random.randint(0, SIZE), "size": random.randint(1, 3), "speed": random.uniform(0.2, 1) }) def make_frame(t): frame = animated.get_frame(t) overlay = frame.copy() for p in particles: y = int((p["y"] - t * 20 * p["speed"]) % SIZE) cv2.circle( overlay, (p["x"], y), p["size"], (255,255,255), -1 ) # cinematic glow blur = cv2.GaussianBlur(overlay, (0,0), 4) final = cv2.addWeighted( overlay, 0.92, blur, 0.15, 0 ) return final final_clip = VideoClip(make_frame, duration=8) # -------------------------------- # تصدير احترافي # -------------------------------- final_clip.write_videofile( "nft_artwork.mp4", fps=30, codec="libx264", bitrate="10000k", audio=False )
