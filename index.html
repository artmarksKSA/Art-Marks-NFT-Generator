import os from PIL import Image, ImageOps # ========================= # الإعدادات # ========================= # مجلد الصور الأصلية INPUT_FOLDER = "images" # مجلد الحفظ OUTPUT_FOLDER = "output" # البرواز الطولي PORTRAIT_FRAME = "frames/frame_portrait.png" # البرواز العرضي LANDSCAPE_FRAME = "frames/frame_landscape.png" # صيغة الحفظ # يفضل PNG لأنه يحافظ على الجودة SAVE_FORMAT = "PNG" # جودة الحفظ إذا استخدمت JPG JPEG_QUALITY = 95 # ========================= # إنشاء مجلد الحفظ إذا لم يكن موجوداً # ========================= if not os.path.exists(OUTPUT_FOLDER): os.makedirs(OUTPUT_FOLDER) # ========================= # التأكد من وجود مجلد الصور # ========================= if not os.path.exists(INPUT_FOLDER): print(f"خطأ: مجلد الصور غير موجود: {INPUT_FOLDER}") exit() # ========================= # امتدادات الصور المدعومة # ========================= SUPPORTED_EXTENSIONS = (".jpg", ".jpeg", ".png", ".webp", ".bmp") # ========================= # قراءة كل الصور من المجلد # ========================= image_files = [ f for f in os.listdir(INPUT_FOLDER) if f.lower().endswith(SUPPORTED_EXTENSIONS) ] if not image_files: print("لا توجد صور داخل مجلد images") exit() # ========================= # معالجة كل صورة # ========================= for file_name in image_files: try: image_path = os.path.join(INPUT_FOLDER, file_name) # فتح الصورة الأصلية وإصلاح الاتجاه التلقائي img = Image.open(image_path) img = ImageOps.exif_transpose(img).convert("RGBA") img_width, img_height = img.size # اختيار البرواز المناسب حسب اتجاه الصورة if img_width > img_height: frame_path = LANDSCAPE_FRAME else: frame_path = PORTRAIT_FRAME # التأكد من وجود البرواز if not os.path.exists(frame_path): print(f"خطأ: ملف البرواز غير موجود: {frame_path}") continue # فتح البرواز frame = Image.open(frame_path).convert("RGBA") # تغيير حجم البرواز ليطابق الصورة تماماً frame_resized = frame.resize((img_width, img_height), Image.LANCZOS) # دمج البرواز فوق الصورة result = Image.alpha_composite(img, frame_resized) # تجهيز اسم الملف الناتج base_name = os.path.splitext(file_name)[0] # حفظ كصورة ثابتة if SAVE_FORMAT.upper() == "PNG": output_path = os.path.join(OUTPUT_FOLDER, f"{base_name}_framed.png") result.save(output_path, "PNG") else: output_path = os.path.join(OUTPUT_FOLDER, f"{base_name}_framed.jpg") result_rgb = result.convert("RGB") result_rgb.save(output_path, "JPEG", quality=JPEG_QUALITY) print(f"تم حفظ الصورة: {output_path}") except Exception as e: print(f"حدث خطأ في الصورة {file_name}: {e}") print("انتهى التنفيذ بنجاح.")
